<head>
    <meta charset="UTF-8">
    <meta http-equiv="=X-UA-Compatible" , content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>별 찍기 과제</title>
</head>

<body>
    <script>
        // 직사각형 만들기

        for (let i = 1; i < 10; i++) { // 'i++ or ++i = i+1', i를  1개씩 증가해 나타낼 때 사용
            for (let j = 0; j < 10; j++) {
                document.write('*'); // document.write는텍스트를 출력할 때 사용하는 함수, 줄 바꿈이 되지 않음
            }
            document.write('<br>'); // <br>은 줄 바꿈 
        }

        // 직삼각형 만들기 1

        for (let i = 0; i < 10; i++) {
            for (let j = 0; j <= i; j++) {
                document.write('*')
            }
            document.write('<br>');
        }

        // 직삼각형 만들기 2

        for (let i = 0; i < 10; i++) {
            for (let j = 10; j > i; j--) { // 'i-- or --i = i-1', i를  1개씩 감소하여 나타낼 때 사용
                document.write('&nbsp'); // &nbsp = 공백, 띄어쓰기
            }

            for (let j = 0; j <= i; j++) {
                document.write('*');
            }
            document.write('<br>');
        }

        // 거꾸로 직삼각형 만들기 1

        for (let i = 0; i < 10; i++) {
            for (let j = 10; j > i; j--) {
                document.write('*');
            }
            document.write('<br>');
        }

        // 거꾸로 직삼각형 만들기 2
        for (let i = 0; i < 10; i++) {
            for (let j = 0; j <= i; j++) {

                document.write('&nbsp');
            }
            for (let j = 10; j > i; j--) {
                document.write('*');
            }
            document.write('<br>');
        }

        // 정삼각형 만들기

        for (let i = 0; i < 10; i++) {
            for (let j = 10; j > i; j--) {

                document.write('&nbsp');
            }
            for (let j = 0; j <= i; j++) {
                document.write('*');
            }
            for (let j = 1; j <= i; j++) {
                document.write('*');
            }
            document.write('<br>');
        }

        document.write('<br>');

        // 거꾸로 정삼각형 만들기

        for (let i = 0; i < 10; i++) {
            for (let j = 0; j < i; j++) {
                document.write('&nbsp');
            }
            for (let j = 9; j > i; j--) {

                document.write('*');
            }
            for (let j = 10; j > i; j--) {
                document.write('*');
            }
            document.write('<br>');
        }


    </script>
    </head>
</body>
